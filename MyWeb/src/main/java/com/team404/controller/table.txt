-- board table
CREATE TABLE FREEBOARD (
    NUM NUMBER(20) PRIMARY KEY,
    WRITER VARCHAR2(20),
    TITLE VARCHAR2(30),
    CONTENT VARCHAR2(200),
    REGDATE DATE DEFAULT SYSDATE,
    MODIDATE DATE DEFAULT SYSDATE
);
CREATE SEQUENCE FB_SEQ START WITH 1 INCREMENT BY 1 NOCACHE;
